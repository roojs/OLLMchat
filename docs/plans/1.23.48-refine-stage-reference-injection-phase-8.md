**Replace** — **`ResultParser.exec_extract(Tool ex)`** — use the shape below (write path fills **`ex.writes`** with **`WriteChange`** fields — **not** **`wc.arguments`**):

```vala
// ResultParser.exec_extract(Tool ex) — summary + document; write path → ex.writes
public bool exec_extract(Tool ex)
{
	if (!this.document.headings.has_key("result-summary")) {
		this.issues += "\nThis task's executor output must include a \"Result summary\" section.";
		return false;
	}
	ex.summary = this.document.headings.get("result-summary");
	ex.document = this.document;
	var definition = ex.parent.skill_manager.fetch(ex.parent);
	if (!definition.tools.contains("write_file")) {
		return true;
	}
	ex.writes.clear();
	if (!this.document.headings.has_key("change-details")) {
		this.issues += "\nWrite executor output must include ## Change details.";
		return false;
	}
	var cd = this.document.headings.get("change-details");
	var doc = this.document;
	var idx = -1;
	for (var i = 0; i < doc.children.size; i++) {
		if (doc.children.get(i) == cd) {
			idx = i;
			break;
		}
	}
	if (idx < 0 || idx + 1 >= doc.children.size) {
		this.issues += "\nChange details: expected a list immediately after the heading.";
		return false;
	}
	var list_node = doc.children.get(idx + 1);
	if (!(list_node is Markdown.Document.List)) {
		this.issues += "\nChange details: first block after heading must be a bullet list.";
		return false;
	}
	var kv = ((Markdown.Document.List) list_node).to_key_map(GLib.CharacterSet.a_2_z);
	var mode = kv.has_key("output_mode")
		? kv.get("output_mode").to_markdown().strip().down() : "";
	string content = "";
	if (mode == "fenced") {
		for (var j = idx + 2; j < doc.children.size; j++) {
			var n = doc.children.get(j);
			if (n is Markdown.Document.Block) {
				var b = (Markdown.Document.Block) n;
				if (b.kind == Markdown.FormatType.FENCED_CODE_QUOTE
						|| b.kind == Markdown.FormatType.FENCED_CODE_TILD) {
					content = b.code_text;
					break;
				}
			}
		}
		if (content == "") {
			this.issues += "\nWrite output: expected a fenced code block after Change details.";
			return false;
		}
	} else if (mode == "next_section") {
		string[] chunks = {};
		for (var j = idx + 2; j < doc.children.size; j++) {
			chunks += doc.children.get(j).to_markdown();
		}
		content = string.joinv("\n\n", chunks).strip();
	} else {
		this.issues += "\nChange details: output_mode must be fenced or next_section.";
		return false;
	}
	var wc = new WriteChange();
	if (kv.has_key("file_path")) {
		wc.file_path = kv.get("file_path").to_markdown().strip();
	}
	if (kv.has_key("ast_path")) {
		wc.ast_path = kv.get("ast_path").to_markdown().strip();
	}
	if (kv.has_key("location")) {
		wc.location = kv.get("location").to_markdown().strip();
	}
	if (kv.has_key("start_line")) {
		wc.start_line = int.parse(kv.get("start_line").to_markdown().strip());
	}
	if (kv.has_key("end_line")) {
		wc.end_line = int.parse(kv.get("end_line").to_markdown().strip());
	}
	if (kv.has_key("complete_file")) {
		var s = kv.get("complete_file").to_markdown().strip().down();
		wc.complete_file = (s == "true" || s == "1");
	}
	if (kv.has_key("overwrite")) {
		var s = kv.get("overwrite").to_markdown().strip().down();
		wc.overwrite = (s == "true" || s == "1");
	}
	wc.content = content;
	ex.writes.add(wc);
	return true;
}
```
