## Coding standards checklist

Copied from **`.cursor/rules/CODING_STANDARDS.md`** — “Checklist for all plans”. If a bullet conflicts with **§ Conventions** in the [parent plan](1.23.40-refine-stage-reference-injection.md), **the standards file wins**.

- **Nullable types**: Design new APIs/properties to avoid nullable types where possible (default objects / flags).
- **Null checks**: Avoid generic null checks; only where the design explicitly requires null (see standards for the narrow exceptions).
- **String interpolation**: No `@"..."` except multi-line usage/help/documentation; use `+` concatenation otherwise.
- **Temporary variables**: No one-use temporaries; no trivial aliases (allowed exception: long property chains **4+** steps).
- **Brace placement**: Line breaks before `{` for namespace/class/method; **inline** `{` for `if`/`for`/`while`/`switch`; never a one-line `if` with body on the same line.
- **`this.` prefix**: Use for instance member access in new/modified Vala.
- **GLib prefix & `using`**: Fully qualify `GLib.*`; do not add `using` for new code.
- **Property initialization**: Prefer `get; set; default = …` (or field defaults), not constructor assignment for those defaults.
- **Line length & breaking**: Break long method calls, concatenations, docblocks, and comments (docblock/comment line length per full standards).
- **StringBuilder**: Avoid `GLib.StringBuilder` unless building strings in loops with **hundreds** of iterations; use `+` and `string.joinv()`.
- **String building in loops**: Never accumulate with `+=` inside a loop; use `string.nfill()` / `replace()`, `string.joinv()`, etc.
- **ArrayList for strings**: Do not use `Gee.ArrayList<string>` only to join; use `string[]` with **`string[] name = {}`** only for empty init (not `new string[0]`).
- **Loops and nesting**: In loops, use **`continue`** to skip cases and keep the body flat; avoid **`else`** and nested **`if`** inside loops.
- **Character looping**: Do not iterate characters unless there is truly no alternative; prefer string methods and `GLib.Regex`.
- **File info try/catch**: Wrap `query_info()`-style calls only when file existence is unknown — not right after a successful read of the same path.
- **Try/catch scope**: Wrap only the minimal expression(s) that throw; keep setup outside `try`.
- **Underscore prefix**: No leading `_` on variables, fields, or properties.
- **`get_*` methods**: Prefer properties or verb-less/action names over `get_foo()` (see standards for exceptions).
- **Docblocks**: Multiline `/** … */` with `@param` / `@return` where relevant; not one-liners for real behavior.
- **Debug / warning messages**: No timestamps or monotonic time in `GLib.debug()` / `GLib.warning()`; no class/method prefixes in the message (file/line come from the log).
- **Gee.HashMap**: Use `.set()` / `.get()` / `.has_key()` / `.unset()` — not `[]`.
- **Gee.ArrayList**: Use `.get()` / `.set()` / `.add()` / `.remove_at()` — not `[]`.
- **Reducing nesting**: Prefer early `return` / `continue`; avoid `else` where flow stays clear; prefer `switch` over long `if` / `else if` chains when editing.
- **SQL table aliases**: Not applicable to this plan’s fences; if you add SQL elsewhere, no table aliases (full names only).
- **Plan Vala fences**: Do **not** use comment-only placeholders (`/* … */` or empty `return "";`) where real logic belongs — either **complete, compilable Vala** in the fence or **prose-only** steps (no fake stub block).

// ResultParser.vala — full replacement of validate_task(); note REFINEMENT branch for extra keys
public void validate_task(Details t, MarkdownPhase phase)
{
	if (t.task_data.has_key("output")) {
		this.issues += "\n" + t.issue_label() + " must not contain Output " +
			"(tasks in the list have no results yet).";
	}
	foreach (var req in required_keys()) {
		if (phase == MarkdownPhase.REFINEMENT && req == "name") {
			continue;
		}
		if (!t.task_data.has_key(req)) {
			this.issues += "\n" + t.issue_label() + " is missing required field: \"" + req + "\". " +
				"Add the missing field to that task (every task must include: " +
				string.joinv(", ", required_keys().to_array()) + ") and resubmit the task list.";
		}
	}
	foreach (var k in t.task_data.keys) {
		if (valid_keys().contains(k)) {
			continue;
		}
		if (phase == MarkdownPhase.REFINEMENT &&
				(k == "shared references" || k == "examination references")) {
			continue;
		}
		this.issues += "\n" + t.issue_label() + " has disallowed field: \"" + k + "\". " +
			"Use only these valid fields: " + string.joinv(", ", valid_keys().to_array()) + ". " +
			"Consider whether the field is wrongly named (fix the name to match a valid field) or " +
			"whether its content belongs in one of the valid fields; then resubmit the task list " +
			"with only the valid fields for each task.";
	}
	if (phase != MarkdownPhase.REFINEMENT && !t.skill_manager.validate(t)) {
		var skill_block = t.task_data.get("skill");
		var skill_name = (skill_block != null) ?
			skill_block.to_markdown().strip() : "";
		this.issues += "\n" + t.issue_label() + " references skill \"" + skill_name + "\", which is not in the available skills list.";
	}
}

// Details.vala — full replacement of build_exec_runs()
public void build_exec_runs()
{
	this.exec_runs = new Gee.ArrayList<Tool>();
	var definition = this.skill_manager.fetch(this);
	var execute_combined = definition.header.has_key("execute-combined") &&
		definition.header.get("execute-combined").strip() != "";
	if (this.tools.size > 0) {
		this.add_exec_runs_for_tools();
		return;
	}
	if (this.exam_references.size > 0) {
		var factory = (OLLMchat.Agent.Factory) this.runner.sr_factory;
		var idx = 0;
		foreach (var exam in this.exam_references) {
			var ex = new Tool(factory, this.session, this, "exam-%d".printf(idx++));
			foreach (var s in this.shared_references) {
				ex.references.add(s);
			}
			ex.references.add(exam);
			var label = exam.title != "" ? exam.title : (exam.href != "" ? exam.href : "ref");
			var target = exam.href != "" ? exam.href : exam.path;
			ex.task_references_markdown = "- **References** [" + label + "](" + target + ")";
			this.exec_runs.add(ex);
		}
		return;
	}
	if (execute_combined) {
		var factory = (OLLMchat.Agent.Factory) this.runner.sr_factory;
		var ex = new Tool(factory, this.session, this, "exec");
		this.add_all_references_to(ex);
		this.exec_runs.add(ex);
		return;
	}
	this.add_exec_runs_for_references();
}

// ResultParser.vala — full replacement of extract_refinement(Details task)
public void extract_refinement(Details task)
{
	if (!this.document.headings.has_key("task")) {
		this.issues += "\n" + "Output must include a \"Task\" section. " +
			"No such section was found. Produce a Task section with a list containing " +
			"the required task keys and optional tool calls.";
		return;
	}
	var found_list = false;
	foreach (var node in this.document.headings.get("task").contents()) {
		if (!(node is Markdown.Document.Block) && !(node is Markdown.Document.List)) {
			continue;
		}
		if (node is Markdown.Document.Block) {
			var block = (Markdown.Document.Block) node;
			if (block.kind != Markdown.FormatType.FENCED_CODE_QUOTE
				&& block.kind != Markdown.FormatType.FENCED_CODE_TILD) {
				continue;
			}
			task.code_blocks.add(block);
			continue;
		}
		found_list = true;
		var list_block = (Markdown.Document.List) node;
		if (list_block.children.size == 0) {
			this.issues += "\n" + "Section \"Task\": the list must contain at least one item.";
			return;
		}
		var refined_map = list_block.to_key_map(GLib.CharacterSet.a_2_z);
		var refined_task = new Details(task.step, refined_map);
		this.validate_task(refined_task, MarkdownPhase.REFINEMENT);
		if (this.issues != "") {
			break;
		}
		task.update_props(refined_map);
		foreach (var link in task.references) {
			task.validate_link(link, MarkdownPhase.REFINEMENT);
		}
		foreach (var link in task.shared_references) {
			task.validate_link(link, MarkdownPhase.REFINEMENT);
		}
		foreach (var link in task.exam_references) {
			task.validate_link(link, MarkdownPhase.REFINEMENT);
		}
		if (task.issues != "") {
			this.issues += "\n" + "Section \"Task\" (References): " + task.issues;
		}
		break;
	}
	if (!found_list) {
		this.issues += "\n" + "Section \"Task\": must contain a list with one task item.";
		return;
	}
	if (!this.document.headings.has_key("tool-calls")) {
		this.document.headings.set("tool-calls", new Markdown.Document.Block(Markdown.FormatType.PARAGRAPH));
	}
	var factory = (OLLMchat.Agent.Factory) this.runner.sr_factory;
	var tool_index = 0;
	foreach (var node in this.document.headings.get("tool-calls").contents()) {
		if (!(node is Markdown.Document.Block)) {
			continue;
		}
		var block = (Markdown.Document.Block) node;
		if (block.kind != Markdown.FormatType.FENCED_CODE_QUOTE
			&& block.kind != Markdown.FormatType.FENCED_CODE_TILD) {
			continue;
		}
		var tool = new Tool(factory, this.runner.session, task, "");
		if (!tool.parse(block)) {
			this.issues += "\n" + tool.issues +
				" Raw block:\n\n```json\n" + block.code_text + "\n```\n\n";
			continue;
		}
		tool_index++;
		tool.tool_call.id = tool.name + "_" + tool_index.to_string();
		if (!tool.validate()) {
			this.issues += "\n" + tool.issues +
				" Raw block:\n\n```json\n" + block.code_text + "\n```\n\n";
			continue;
		}
		task.tools.add(tool);
	}
	var definition = task.skill_manager.fetch(task);
	if (definition.tools.size > 0 && !definition.tools.contains("write_file")) {
		if (task.tools.size == 0) {
			this.issues += "\n" + "Tool refinement (skill lists tools other than write_file): output must include at least one fenced JSON tool call in ## Tool Calls.";
		}
	} else if (!definition.tools.contains("write_file")) {
		if (task.references.size == 0 && task.shared_references.size == 0
				&& task.exam_references.size == 0 && task.tools.size == 0) {
			this.issues += "\n" + "Reference refinement must provide at least one of: references, shared references, examination references, or ## Tool Calls.";
		}
	}
}
