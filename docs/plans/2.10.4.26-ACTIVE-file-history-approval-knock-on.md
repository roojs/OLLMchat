### 2. `libocfiles/V2/FileHistory.vala` â€” **Add** (new file)

**Add** â€” same as **2.10.4.25** Â§5 wire contract; **`revert()`** merges **`File`** row then **`read()`** (**G-2**):

```vala
namespace OLLMfiles
{
	/**
	 * File change history row (V2 approve/revert RPC).
	 */
	public class FileHistory : Object
	{
		public static void rpc_register()
		{
			OLLMrpc.register("FileHistory", typeof(FileHistory));
		}

		public weak ProjectManager manager { get; set; }
		public int64 id { get; set; default = 0; }
		public string path { get; set; default = ""; }

		public async void approve()
		{
			yield this.manager.rpc.call(new OLLMrpc.Request() {
				method = "FileHistory.approve",
				param = new OLLMfilesd.FileParams() {
					path = this.path,
					id = this.id
				}
			});
		}

		public async void revert()
		{
			var response = yield this.manager.rpc.call(
				new OLLMrpc.Request() {
					method = "FileHistory.revert",
					param = new OLLMfilesd.FileParams() {
						path = this.path,
						id = this.id
					}
				}
			);
			if (response.error != null) {
				return;
			}
			var cached = this.manager.file_cache.get(this.path);
			if (cached == null && response.result is File) {
				this.manager.file_cache.set(
					this.path,
					(File) response.result
				);
				cached = this.manager.file_cache.get(this.path);
			}
			if (cached == null) {
				return;
			}
			cached.manager = this.manager;
			if (response.result is File) {
				cached.copy_from((File) response.result, {
					"manager",
					"buffer",
					"parent"
				});
			}
			yield cached.read();
		}
	}
}
```

**â„ąď¸Ź** **`rpc_register`** line omitted if user picks **G-3 Option B**. **`cached`** path assumes **G-4 Option A** for list rows; adjust if user picks **G-4 Option B**.
