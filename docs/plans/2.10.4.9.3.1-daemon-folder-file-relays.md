### 1. `ollmfilesd/File.vala` — `File.fetch` relay

**Why:** Backs client `Folder.fetch_file` ([`2.10.4.9.1`](2.10.4.9.1-design-project-files-and-client-api.md) §4).

**Where:** `File` class — signal list and `construct` block (mirror existing `rpc_read`).

**Depends on:** parent §1 `FileParams.project_path`.

#### Add — `rpc_fetch` signal

Placement: after `rpc_read` signal declaration.

```vala
		public signal void rpc_fetch(OLLMrpc.Request request);
```

#### Add — `rpc_fetch` handler in `construct`

Placement: immediately after the closing `});` of the `rpc_read.connect` handler.

```vala
			this.rpc_fetch.connect((request) => {
				var p = (FileParams) request.param;
				File? source = null;
				foreach (var entry in this.manager.projects.project_map.entries) {
					if (entry.value.path != p.project_path
						|| !entry.value.is_project) {
						continue;
					}
					var project_file = entry.value.project_files.child_map.get(
						p.path
					);
					if (project_file != null) {
						source = project_file.file;
					}
					break;
				}
				if (source == null
					&& this.manager.active_project != null
					&& this.manager.active_project.path == p.project_path) {
					var project_file =
						this.manager.active_project.project_files.child_map.get(
							p.path
						);
					if (project_file != null) {
						source = project_file.file;
					}
				}
				if (source == null) {
					request.reply(new OLLMrpc.Response() {
						id = request.id,
						result_type = "File"
					});
					return;
				}
				var row = new File(this.manager);
				row.copy_from(source, {"manager", "buffer", "parent"});
				row.last_modified = source.mtime_on_disk();
				request.reply(new OLLMrpc.Response() {
					id = request.id,
					result = row,
					result_type = "File"
				});
			});
```


#### Add — `construct` block with handlers

Placement: immediately after the four signal declarations above.

```vala
		construct
		{
			this.rpc_fetch.connect((request) => {
				var path = ((FolderParams) request.param).path;
				var folder = this.manager.get_folder_at_path(path);
				request.reply(new OLLMrpc.Response() {
					id = request.id,
					result = folder,
					result_type = "Folder"
				});
			});
			this.rpc_contains_folder.connect((request) => {
				var p = (FolderParams) request.param;
				var project = this.find_project_root(p.project_path);
				var found = false;
				if (project != null) {
					found = project.project_files.folder_map.has_key(p.path);
				}
				request.reply(new OLLMrpc.Response() {
					id = request.id,
					msg = found ? "true" : "false"
				});
			});
			this.rpc_fetch_file_list.connect((request) => {
				var path = ((FolderParams) request.param).path;
				var project = this.find_project_root(path);
				var list = new Gee.ArrayList<GLib.Object>();
				if (project != null) {
					foreach (var project_file in project.project_files) {
						list.add(project_file.file);
					}
				}
				request.reply(new OLLMrpc.Response() {
					id = request.id,
					result = list,
					result_type = "File",
					is_array = true
				});
			});
			this.rpc_fetch_pending_approvals.connect((request) => {
				var path = ((FolderParams) request.param).path;
				var project = this.find_project_root(path);
				var list = new Gee.ArrayList<GLib.Object>();
				if (project != null) {
					project.project_files.review_files.refresh();
					for (uint i = 0; i < project.project_files.review_files.get_n_items(); i++) {
						var pf = project.project_files.review_files.get_item(i)
							as ProjectFile;
						if (pf != null) {
							list.add(pf.file);
						}
					}
				}
				request.reply(new OLLMrpc.Response() {
					id = request.id,
					result = list,
					result_type = "File",
					is_array = true
				});
			});
		}
```

#### Add — project lookup helper

Placement: end of `Folder` class, before its closing `}` (after `build_roots`).

```vala
		/**
		 * Resolve a project root row by path for RPC relays on this class.
		 *
		 * @param project_path Project root absolute path
		 * @return The project folder, or null if not loaded
		 */
		private Folder? find_project_root(string project_path)
		{
			if (project_path == "") {
				return null;
			}
			foreach (var entry in this.manager.projects.project_map.entries) {
				if (entry.value.path == project_path && entry.value.is_project) {
					return entry.value;
				}
			}
			if (this.manager.active_project != null
				&& this.manager.active_project.path == project_path) {
				return this.manager.active_project;
			}
			return null;
		}
```
