### 3. `libollmchat/Settings/SearchResults.vala` — new file

**Why:** Session + debounce + `ListModel` of `OllamaWeb.Model` (no duplicate app types).

**Depends on:** §1.

#### Add — New file (full contents).

```vala
namespace OLLMchat.Settings
{
	public class SearchResults : Object, GLib.ListModel
	{
		private const uint SEARCH_DEBOUNCE_MS = 2000;

		public string data_dir { get; construct; }

		public OllamaWeb.Search.Session session {
			get;
			private set;
			default = new OllamaWeb.Search.Session();
		}

		private Gee.ArrayList<OllamaWeb.Model> store {
			get;
			set;
			default = new Gee.ArrayList<OllamaWeb.Model>();
		}

		private uint debounce_id;
		private string? pending_query;

		public SearchResults(string data_dir)
		{
			Object(data_dir: data_dir);
			this.session.model_dir = GLib.Path.build_filename(this.data_dir, "ollamaweb-models");
		}

		public Type get_item_type()
		{
			return typeof(OllamaWeb.Model);
		}

		public uint get_n_items()
		{
			return this.store.size;
		}

		public Object? get_item(uint position)
		{
			if (position >= this.store.size) {
				return null;
			}
			return this.store[(int)position];
		}

		public void queue_search(string query)
		{
			if (this.debounce_id != 0) {
				GLib.Source.remove(this.debounce_id);
				this.debounce_id = 0;
			}
			var stripped = query.strip();
			this.pending_query = stripped;
			if (stripped == "") {
				this.clear_results();
				this.session.cancel();
				return;
			}
			this.debounce_id = GLib.Timeout.add(SEARCH_DEBOUNCE_MS, () => {
				this.debounce_id = 0;
				this.run_search.begin(this.pending_query);
				return false;
			});
		}

		public void cancel()
		{
			if (this.debounce_id != 0) {
				GLib.Source.remove(this.debounce_id);
				this.debounce_id = 0;
			}
			this.pending_query = null;
			this.session.cancel();
		}

		private void clear_results()
		{
			var n = this.store.size;
			if (n == 0) {
				return;
			}
			this.store.clear();
			this.items_changed(0, n, 0);
		}

		private async void run_search(string? query)
		{
			if (query == null || query.strip() == "") {
				return;
			}
			var q = query.strip();
			try {
				var hits = yield this.session.search(q, OllamaWeb.Search.Category.NONE);
				if (this.pending_query != q) {
					return;
				}
				this.replace_hits(hits);
				if (this.session.refine_queue.size > 0) {
					yield this.session.refine();
					if (this.pending_query != q) {
						return;
					}
					foreach (var row in this.store) {
						row.notify_property("list_markup");
					}
				}
			} catch (GLib.Error e) {
				GLib.warning("ollama.com search failed: %s", e.message);
			}
		}

		private void replace_hits(Gee.ArrayList<OllamaWeb.Model> hits)
		{
			var n_old = this.store.size;
			this.store.clear();
			foreach (var hit in hits) {
				this.store.add(hit);
			}
			this.items_changed(0, n_old, this.store.size);
		}

	}
}
```
