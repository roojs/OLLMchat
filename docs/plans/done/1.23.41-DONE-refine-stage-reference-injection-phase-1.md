## Coding standards checklist

Copied from **`.cursor/rules/CODING_STANDARDS.md`** — “Checklist for all plans”. If a bullet conflicts with **§ Conventions** in the [parent plan](1.23.40-refine-stage-reference-injection.md), **the standards file wins**.

- **Nullable types**: Design new APIs/properties to avoid nullable types where possible (default objects / flags).
- **Null checks**: Avoid generic null checks; only where the design explicitly requires null (see standards for the narrow exceptions).
- **String interpolation**: No `@"..."` except multi-line usage/help/documentation; use `+` concatenation otherwise.
- **Temporary variables**: No one-use temporaries; no trivial aliases (allowed exception: long property chains **4+** steps).
- **Brace placement**: Line breaks before `{` for namespace/class/method; **inline** `{` for `if`/`for`/`while`/`switch`; never a one-line `if` with body on the same line.
- **`this.` prefix**: Use for instance member access in new/modified Vala.
- **GLib prefix & `using`**: Fully qualify `GLib.*`; do not add `using` for new code.
- **Property initialization**: Prefer `get; set; default = …` (or field defaults), not constructor assignment for those defaults.
- **Line length & breaking**: Break long method calls, concatenations, docblocks, and comments (docblock/comment line length per full standards).
- **StringBuilder**: Avoid `GLib.StringBuilder` unless building strings in loops with **hundreds** of iterations; use `+` and `string.joinv()`.
- **String building in loops**: Never accumulate with `+=` inside a loop; use `string.nfill()` / `replace()`, `string.joinv()`, etc.
- **ArrayList for strings**: Do not use `Gee.ArrayList<string>` only to join; use `string[]` with **`string[] name = {}`** only for empty init (not `new string[0]`).
- **Loops and nesting**: In loops, use **`continue`** to skip cases and keep the body flat; avoid **`else`** and nested **`if`** inside loops.
- **Character looping**: Do not iterate characters unless there is truly no alternative; prefer string methods and `GLib.Regex`.
- **File info try/catch**: Wrap `query_info()`-style calls only when file existence is unknown — not right after a successful read of the same path.
- **Try/catch scope**: Wrap only the minimal expression(s) that throw; keep setup outside `try`.
- **Underscore prefix**: No leading `_` on variables, fields, or properties.
- **`get_*` methods**: Prefer properties or verb-less/action names over `get_foo()` (see standards for exceptions).
- **Docblocks**: Multiline `/** … */` with `@param` / `@return` where relevant; not one-liners for real behavior.
- **Debug / warning messages**: No timestamps or monotonic time in `GLib.debug()` / `GLib.warning()`; no class/method prefixes in the message (file/line come from the log).
- **Gee.HashMap**: Use `.set()` / `.get()` / `.has_key()` / `.unset()` — not `[]`.
- **Gee.ArrayList**: Use `.get()` / `.set()` / `.add()` / `.remove_at()` — not `[]`.
- **Reducing nesting**: Prefer early `return` / `continue`; avoid `else` where flow stays clear; prefer `switch` over long `if` / `else if` chains when editing.
- **SQL table aliases**: Not applicable to this plan’s fences; if you add SQL elsewhere, no table aliases (full names only).
- **Plan Vala fences**: Do **not** use comment-only placeholders (`/* … */` or empty `return "";`) where real logic belongs — either **complete, compilable Vala** in the fence or **prose-only** steps (no fake stub block).

```vala
// Details.vala — full replacement of to_markdown(MarkdownPhase phase)
public string to_markdown(MarkdownPhase phase)
{
	string[] order = {
		"name",
		"what is needed",
		"skill",
		"references",
		"shared references",
		"examination references",
		"expected output",
		"requires user approval",
		"output"
	};
	var ret = "";
	for (var i = 0; i < order.length; i++) {
		var key = order[i];
		switch (key) {
			case "references":
			case "shared references":
			case "examination references":
				if (phase == MarkdownPhase.REFINE_COMPLETED) {
					continue;
				}
				break;
			case "output":
				if ((phase != MarkdownPhase.LIST &&
						phase != MarkdownPhase.REFINE_COMPLETED)
						|| !this.exec_done) {
					continue;
				}
				// Executor output: always include the post-exec result summary (Result summary body).
				ret += "#### Task result\n\n";
				ret += this.task_post_exec_summary.to_markdown_with_content()
					.strip() + "\n\n";
				// Other top-level headings in the output doc: bullet list of [title](#gfm-slug) for in-doc navigation.
				if (this.task_output_document.header_list.size == 0) {
					continue;
				}
				string[] section_links = {};
				foreach (var slug in this.task_output_document.header_list) {
					var hb = this.task_output_document.headings.get(slug);
					var title = hb != null ? hb.text_content().strip() : slug;
					if (title == "") {
						title = slug;
					}
					section_links += "- [" + title + "](#" + slug + ")";
				}
				ret += "**Sections in this output:**\n\n"
					+ string.joinv("\n", section_links) + "\n\n";
				continue;
			default:
				break;
		}
		if (!this.task_data.has_key(key)) {
			continue;
		}
		var block = this.task_data.get(key);
		if (block.children.size == 0) {
			continue;
		}
		ret += "- **" + key.substring(0, 1).up() + key.substring(1) + "** "
			+ block.to_markdown() + "\n";
	}
	if (phase == MarkdownPhase.REFINE_COMPLETED || phase == MarkdownPhase.POST_EXEC ||
			this.tools.size == 0) {
		return ret;
	}
	ret += "\n## Tool Calls\n\n";
	foreach (var tool in this.tools) {
		var obj = new Json.Object();
		obj.set_string_member("name", tool.name);
		if (tool.tool_call != null && tool.tool_call.id != "") {
			obj.set_string_member("id", tool.tool_call.id);
		}
		if (tool.arguments != null) {
			obj.set_object_member("arguments", tool.arguments);
		}
		var node = new Json.Node(Json.NodeType.OBJECT);
		node.set_object(obj);
		var gen = new Json.Generator();
		gen.set_root(node);
		ret += "```json\n" + gen.to_data(null) + "\n```\n\n";
	}
	return ret;
}
```

```vala
// List.vala — full replacement of to_key_map() (signatures + behaviour contract)
public Gee.Map<string, Block> to_key_map(string key_mode = "")
{
	var result = new Gee.HashMap<string, Block>();
	foreach (var node in this.children) {
		var item = node as ListItem;
		if (item == null) {
			continue;
		}
		var v = new Block(FormatType.PARAGRAPH);
		var k = item.key_value(v).strip();
		if (k == "") {
			continue;
		}
		k = key_mode == GLib.CharacterSet.a_2_z ? k.down() : k;
		result.set(k, v);
	}
	return result;
}
```

```vala
// Details.vala — full replacement of fill_task_data()
/**
 * Ensure expected task_data keys exist (placeholder block if missing), then refresh
 * fields derived from task_data (e.g. references, shared_references, exam_references).
 * Called from the ctor and update_props after task_data changes — the name alone
 * does not spell out that dual role.
 */
private void fill_task_data()
{
	this.requires_user_approval = this.task_data.has_key("requires user approval");
	var empty = new Markdown.Document.Block(Markdown.FormatType.PARAGRAPH);
	string[] keys_to_fill = {
		"what is needed",
		"skill",
		"references",
		"expected output",
		"shared references",
		"examination references" };
	foreach (var key in keys_to_fill) {
		if (!this.task_data.has_key(key)) {
			this.task_data.set(key, empty);
		}
		switch (key) {
			case "references":
				this.references = this.task_data.get(key).links();
				break;
			case "shared references":
				this.shared_references = this.task_data.get(key).links();
				break;
			case "examination references":
				this.exam_references = this.task_data.get(key).links();
				break;
			default:
				break;
		}
	}
}
```

```vala
// Runner.vala — private static helper (place with other private methods in Runner)
private static string abbreviate_markdown_lines(string md)
{
	if (md == "") {
		return "";
	}
	string[] lines = md.split("\n");
	if (lines.length <= 10) {
		return md;
	}
	string[] head = {};
	for (var li = 0; li < 10; li++) {
		head += lines[li];
	}
	return string.joinv("\n", head) + "\n\n_(abbreviated)_\n";
}
```
